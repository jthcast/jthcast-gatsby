```js
const getLCM = (n, m) => {
  const getGCD = (n, m) => {
    return m ? getGCD(m, n % m) : n;
  };

  return (n * m) / getGCD(n, m);
};
```

## 입력

const n = 2;

const m = 4;

## 출력

4
