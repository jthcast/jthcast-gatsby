```js
class Queue {
  constructor() {
    this.array = [];
  }

  add(item) {
    this.array.push(item);
  }

  peek() {
    return this.array[0];
  }

  poll() {
    return this.array.shift();
  }

  isEmpty() {
    return this.array.length === 0;
  }
}
```
